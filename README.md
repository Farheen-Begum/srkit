+ i ---> for insert
+ ctrl+c ---> :wq--->saves and exits
+ rm filename ---> removes file
+ git add filename
+ git add . or git add --all
+ git status

+  git config --global user.name

+  git config --global user.email
+ git config --global --unset user.email "email"
